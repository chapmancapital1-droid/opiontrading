class KnowledgeBase:
    """Structured brain - loads/saves catalog, supports semantic queries later."""
    def __init__(self, catalog_path: str = "ota_knowledge_catalog.json"):
        self.catalog_path = catalog_path
        self.catalog = self._load_catalog()
        logger.info("KnowledgeBase initialized with %d categories.", len(self.catalog))
    
    def _load_catalog(self) -> Dict:
        if os.path.exists(self.catalog_path):
            with open(self.catalog_path, 'r') as f:
                return json.load(f)
        # Seed with core structure
        return {
            "Core Foundations": {},
            "Numerical Methods": {"Monte Carlo": {}},
            "Trading Strategies": {},
            "Risk Management": {},
            "Python & Automation": {},
            "ML/AI Integration": {},
            "Data Pipelines": {},
            "Edge Cases": {},
            "Source References": []
        }
    
    def save(self):
        with open(self.catalog_path, 'w') as f:
            json.dump(self.catalog, f, indent=2)
        logger.info("Knowledge catalog saved.")
    
    def add_knowledge(self, category: str, subsection: str, content: Dict, source: str = "ingested"):
        if category not in self.catalog:
            self.catalog[category] = {}
        if subsection not in self.catalog[category]:
            self.catalog[category][subsection] = []
        entry = {"content": content, "source": source, "timestamp": datetime.now().isoformat()}
        self.catalog[category][subsection].append(entry)
        self.save()
        logger.info(f"Added knowledge to {category} > {subsection} from {source}")

class OptionPricing:
    """Pricing engine with BS + Monte Carlo."""
    @staticmethod
    def black_scholes_call(S: float, X: float, T: float, r: float, sigma: float) -> float:
        if T <= 0:
            return max(S - X, 0.0)
        d1 = (np.log(S / X) + (r + 0.5 * sigma**2) * T) / (sigma * np.sqrt(T))
        d2 = d1 - sigma * np.sqrt(T)
        return S * norm.cdf(d1) - X * np.exp(-r * T) * norm.cdf(d2)
    
    @staticmethod
    def monte_carlo_european_call(S0: float, X: float, T: float, r: float, sigma: float,
                                  num_sim: int = 50000, steps: int = 252) -> Tuple[float, float]:
        dt = T / steps
        drift = (r - 0.5 * sigma**2) * dt
        vol = sigma * np.sqrt(dt)
        np.random.seed(42)
        Z = np.random.normal(0, 1, (num_sim, steps))
        log_S = np.cumsum(drift + vol * Z, axis=1)
        S_T = S0 * np.exp(log_S[:, -1])
        payoffs = np.maximum(S_T - X, 0)
        price = np.exp(-r * T) * np.mean(payoffs)
        std_err = np.exp(-r * T) * np.std(payoffs) / np.sqrt(num_sim)
        return price, std_err

class DecisionEngine:
    """Trade recommendation logic using knowledge + pricing."""
    def __init__(self, kb: KnowledgeBase, pricing: OptionPricing):
        self.kb = kb
        self.pricing = pricing
    
    def recommend(self, S0: float, X: float, T: float, r: float, sigma: float,
                  market_price: Optional[float] = None, news_sentiment: float = 0.0) -> Dict:
        bs = self.pricing.black_scholes_call(S0, X, T, r, sigma)
        mc, err = self.pricing.monte_carlo_european_call(S0, X, T, r, sigma)
        fair = (bs + mc) / 2 * (1 + 0.1 * news_sentiment)  # News boost example
        edge = fair - (market_price or fair)
        
        rec = "HOLD"
        size = 0
        if edge > 0.05 * fair:
            rec = "BUY CALL"
            # Risk sizing logic here (expand from ingested risk rules)
            size = 1  # Placeholder - calculate from portfolio risk tolerance
        elif edge < -0.05 * fair:
            rec = "SELL / CONSIDER PUT"
        
        return {
            "timestamp": datetime.now().isoformat(),
            "fair_value": round(fair, 4),
            "edge": round(edge, 4),
            "recommendation": rec,
            "size": size,
            "confidence": min(1.0, abs(edge) / (err + 0.01)),
            "sources": ["Black-Scholes + Monte Carlo (Crack Ch.5,8)", "News adjustment"]
        }

class PortfolioManager:
    """Simple portfolio & risk tracking."""
    def __init__(self, initial_cash: float = 50000, risk_per_trade: float = 0.02):
        self.cash = initial_cash
        self.positions = []
        self.risk_per_trade = risk_per_trade
        logger.info("PortfolioManager ready with $%.2f cash.", initial_cash)
    
    def execute(self, rec: Dict):
        if "BUY" in rec["recommendation"] and rec["size"] > 0:
            cost = rec["size"] * rec["fair_value"] * 100
            if cost <= self.cash:
                self.cash -= cost
                self.positions.append(rec)
                logger.info("Executed trade. Cash remaining: $%.2f", self.cash)

class NERDCOMMAND_OTA:
    """Main agent orchestrator."""
    def __init__(self):
        self.kb = KnowledgeBase()
        self.ingestor = KnowledgeIngestion(self.kb)
        self.pricing = OptionPricing()
        self.decision = DecisionEngine(self.kb, self.pricing)
        self.portfolio = PortfolioManager()
        logger.info("NERDCOMMAND OTA Brain fully initialized.")
    
    def ingest_new_book(self, pdf_path: str, book_name: str):
        return self.ingestor.ingest_pdf(pdf_path, book_name)
    
    def analyze_option(self, **params):
        rec = self.decision.recommend(**params)
        self.portfolio.execute(rec)
        logger.info("Analysis complete: %s", rec["recommendation"])
        return rec
    
    def run_demo(self):

                  "market_price": 10.5, "news_sentiment": 0.15}
        return self.analyze_option(**params)
